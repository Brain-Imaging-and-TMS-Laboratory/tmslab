<!--
  Auto-discovery: every .jpg/.jpeg/.png/.webp file in
  /assets/img/gallery/ is picked up automatically at build time.
  Drop a file in → it appears on next rebuild. Remove a file →
  it disappears. No YAML to maintain.

  Captions are derived from the filename: underscores and hyphens
  become spaces, then each word is capitalized.

  Filtered out: responsive variants generated by jekyll-imagemagick
  (basenames ending in -480 / -800 / -1400). Only the originals show.
-->

<div class="tmslab-gallery-header">
  <p class="tmslab-gallery-subtitle">A collection of moments from the lab — retreats, events, conference travel, and the quiet work between experiments.</p>
</div>

{%- assign gallery_images = site.static_files | where_exp: "f", "f.path contains '/assets/img/gallery/'" -%}
{%- assign gallery_images = gallery_images | where_exp: "f", "f.extname == '.jpg' or f.extname == '.jpeg' or f.extname == '.png' or f.extname == '.webp'" -%}
{%- assign gallery_images = gallery_images | sort: "basename" | reverse -%}

{%- comment -%}
  Count the displayable originals (skip imagemagick variants) so the
  empty-state branch fires correctly when only variants would remain.
{%- endcomment -%}
{%- assign visible_count = 0 -%}
{%- for image in gallery_images -%}
  {%- assign segs = image.basename | split: "-" -%}
  {%- assign last = segs | last -%}
  {%- unless last == "480" or last == "800" or last == "1400" -%}
    {%- assign visible_count = visible_count | plus: 1 -%}
  {%- endunless -%}
{%- endfor -%}

{% if visible_count == 0 %}
  <p class="tmslab-gallery-empty">Photos coming soon.</p>
{% else %}
  <div class="tmslab-gallery-grid" id="tmslab-gallery">
    {%- for image in gallery_images -%}
      {%- assign segs = image.basename | split: "-" -%}
      {%- assign last_seg = segs | last -%}
      {%- unless last_seg == "480" or last_seg == "800" or last_seg == "1400" -%}

        {%- assign with_spaces = image.basename | replace: "_", " " | replace: "-", " " -%}
        {%- assign words = with_spaces | split: " " -%}
        {%- assign caption = "" -%}
        {%- for w in words -%}
          {%- if w != "" -%}
            {%- assign cap_w = w | capitalize -%}
            {%- if caption == "" -%}
              {%- assign caption = cap_w -%}
            {%- else -%}
              {%- assign caption = caption | append: " " | append: cap_w -%}
            {%- endif -%}
          {%- endif -%}
        {%- endfor -%}

        <figure class="tmslab-gallery-item" data-caption="{{ caption | strip }}" data-full="{{ image.path | relative_url }}">
          <img src="{{ image.path | relative_url }}" alt="{{ caption | strip }}" loading="lazy">
          <figcaption class="tmslab-gallery-caption">{{ caption | strip }}</figcaption>
        </figure>

      {%- endunless -%}
    {%- endfor -%}
  </div>

  <script src="{{ '/assets/js/gallery-lightbox.js' | relative_url }}" defer></script>
{% endif %}
